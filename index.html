// QPRIME
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int L, R, digit;
    cin >> L >> R >> digit;

    vector<bool> isPrime(R - L + 1, true);

    for (int i = 2; i * i <= R; ++i) {
        for (int j = max(i * i, (L + i - 1) / i * i); j <= R; j += i) {
            isPrime[j - L] = false;
        }
    }
    if (1 >= L) {
        isPrime[1 - L] = false;
    }

    int ans = 0;
    for (int x = L; x <= R; ++x) {
        if (isPrime[x - L]) {
            int t = x;
            while(t > 0) {
                if(t % 10 == digit) {
                    ans++;
                    break;
                }
                t /= 10;
            }
        }
    }
    cout << ans;
    return 0;
}

#include <bits/stdc++.h>
using namespace std;
const int maxn = 11;
const int maxf = 1001;
const int inf = 999999;

int main() {
    
    int c, n;
    cin >> c >> n;
    int a[maxn];
    int f[maxn][maxf];
    for(int i = 1; i <= n; i++) {
        cin >> a[i];
    }
      
    for(int i = 0; i <= n; i++) {
        for(int j = 0; j <= c; j++) {
            if(j == 0) f[i][j] = 0;
            else if(i == 0) f[i][j] = inf;
            else if(j < a[i]) f[i][j] = f[i-1][j];
            else f[i][j] = min(f[i-1][j], f[i][j - a[i]] + 1);
        }
    }
  
    cout << f[n][c];
      
    return 0;
}
            
        
            
//input: c, n, n coin values
//output: minimum number of coins

#include <iostream>
#include <vector>
using namespace std;

int main() {
    int c, n;
    cin >> c >> n;
    vector<int> coins(n);
    for (int i = 0; i < n; i++) {
        cin >> coins[i];
    }

    vector<int> f(c+1);
    f[0] = 0;
    for (int i = 1; i <= c; i++) {
        f[i] = c+1;
        for (int coin: coins) {
            if (i-coin >= 0 && f[i] > f[i-coin]+1) {
                f[i] = f[i-coin]+1;
            }
        }
    }
    cout << f[c] << endl;

    return 0;
}
